# 登录系统设计文档

## 一、登录流程设计

### 1. 首次登录完整流程（学生）

```
用户打开小程序
    ↓
检查本地是否有登录信息
    ↓
【没有】→ 显示登录引导页（pages/login/login）
    ↓
点击"微信登录"按钮
    ↓
调用 wx.login() 获取 code
    ↓
调用云函数 login(code)
    ↓
【返回 openid + 用户基本信息】
    ↓
检查该用户是否已选择角色
    ↓
【未选择】→ 跳转到角色选择页（pages/role-select/role-select）
    ↓
用户选择"我是学生"
    ↓
显示年级选择器（七上～九下）
    ↓
用户选择年级
    ↓
调用云函数 updateUserRole({ role: "student", grade: "七上" })
    ↓
保存用户信息到本地
    ↓
【可选】检测本地是否有学习进度数据
    ↓
【有】→ 提示"是否上传到云端？"
    ↓
【用户同意】→ 调用云函数 syncProgress(本地数据)
    ↓
跳转到首页（pages/index/index）
```

### 2. 首次登录完整流程（教师）

```
用户打开小程序
    ↓
... (前面步骤同学生)
    ↓
【角色选择页】用户选择"我是教师"
    ↓
显示教师信息填写表单
    ├─ 学校名称
    ├─ 真实姓名
    ├─ 手机号（可选）
    └─ 教师资格证照片（可选）
    ↓
用户填写并提交
    ↓
调用云函数 updateUserRole({
  role: "teacher",
  teacherInfo: { school, realName, ... }
})
    ↓
【后端】设置 certificationStatus = "pending"
    ↓
显示"认证审核中"提示（教师端首次体验可以先不审核）
    ↓
保存用户信息到本地
    ↓
跳转到教师端首页（pages/teacher-home/teacher-home）
```

### 3. 再次登录流程

```
用户打开小程序
    ↓
检查本地是否有登录信息
    ↓
【有】→ 读取 openid 和 role
    ↓
验证 openid 是否有效（可选，定期验证）
    ↓
根据 role 跳转到对应首页
    ├─ student → pages/index/index
    └─ teacher → pages/teacher-home/teacher-home
    ↓
【可选】后台拉取云端数据更新
```

### 4. 自动登录 vs 手动登录

**自动登录（推荐）：**
- 用户打开小程序自动检测登录状态
- 如果本地有 openid 且未过期，直接进入首页
- 静默更新云端数据

**手动登录：**
- 用户主动点击"登录"按钮
- 适用于退出登录后重新登录的场景

### 1. app.js - 全局登录检测

```javascript
// app.js
App({
  globalData: {
    userInfo: null,      // 用户信息
    openid: null,        // 用户openid
    role: null,          // 用户角色
    isLoggedIn: false    // 是否已登录
  },

  onLaunch() {
    // 初始化云开发环境
    if (wx.cloud) {
      wx.cloud.init({
        env: 'your-env-id',  // 云开发环境ID
        traceUser: true
      });
    }

    // 检查登录状态
    this.checkLoginStatus();
  },

  /**
   * 检查登录状态
   */
  checkLoginStatus() {
    const loginInfo = wx.getStorageSync('userLoginInfo');

    if (loginInfo && loginInfo.openid) {
      // 有登录信息，自动登录
      this.globalData.userInfo = loginInfo;
      this.globalData.openid = loginInfo.openid;
      this.globalData.role = loginInfo.role;
      this.globalData.isLoggedIn = true;

      // 可选：验证 openid 是否仍然有效
      this.verifyLogin();
    } else {
      // 没有登录信息，跳转到登录页
      this.navigateToLogin();
    }
  },

  /**
   * 验证登录状态（可选）
   */
  verifyLogin() {
    // 调用云函数验证 openid
    // 如果失效，清除本地数据并重新登录
  },

  /**
   * 跳转到登录页
   */
  navigateToLogin() {
    wx.redirectTo({
      url: '/pages/login/login'
    });
  },

  /**
   * 登录成功后保存用户信息
   */
  onLoginSuccess(userInfo) {
    this.globalData.userInfo = userInfo;
    this.globalData.openid = userInfo.openid;
    this.globalData.role = userInfo.role;
    this.globalData.isLoggedIn = true;

    wx.setStorageSync('userLoginInfo', {
      openid: userInfo.openid,
      role: userInfo.role,
      nickName: userInfo.userInfo.nickName,
      avatarUrl: userInfo.userInfo.avatarUrl,
      lastLoginAt: new Date().toISOString()
    });
  },

  /**
   * 退出登录
   */
  logout() {
    this.globalData.userInfo = null;
    this.globalData.openid = null;
    this.globalData.role = null;
    this.globalData.isLoggedIn = false;

    wx.removeStorageSync('userLoginInfo');
    wx.redirectTo({
      url: '/pages/login/login'
    });
  }
});
```

### 2. login 页面核心逻辑

```javascript
// pages/login/login.js
Page({
  data: {
    loading: false
  },

  /**
   * 微信登录
   */
  async onWechatLogin() {
    this.setData({ loading: true });

    try {
      // 1. 获取微信登录凭证
      const loginRes = await wx.login();
      const code = loginRes.code;

      // 2. 调用云函数登录
      const cloudRes = await wx.cloud.callFunction({
        name: 'login',
        data: { code }
      });

      const { openid, userInfo, role } = cloudRes.result;

      // 3. 检查用户是否已选择角色
      if (!role) {
        // 未选择角色，跳转到角色选择页
        wx.navigateTo({
          url: '/pages/role-select/role-select'
        });
      } else {
        // 已选择角色，保存用户信息并进入首页
        const app = getApp();
        app.onLoginSuccess({
          openid,
          userInfo,
          role
        });

        // 根据角色跳转到不同首页
        const homePage = role === 'teacher'
          ? '/pages/teacher-home/teacher-home'
          : '/pages/index/index';

        wx.redirectTo({ url: homePage });
      }
    } catch (err) {
      console.error('登录失败:', err);
      wx.showToast({
        title: '登录失败，请重试',
        icon: 'none'
      });
    } finally {
      this.setData({ loading: false });
    }
  }
});
```

## 四、云函数实现

### 1. login 云函数

```javascript
// cloudfunctions/login/index.js
const cloud = require('wx-server-sdk');
cloud.init({ env: cloud.DYNAMIC_CURRENT_ENV });

const db = cloud.database();

exports.main = async (event, context) => {
  const { code } = event;
  const wxContext = cloud.getWXContext();
  const openid = wxContext.OPENID;

  try {
    // 查询用户是否已存在
    const userRes = await db.collection('users').where({
      _openid: openid
    }).get();

    let userData;

    if (userRes.data.length === 0) {
      // 新用户，创建记录（role 为空，待选择）
      const createRes = await db.collection('users').add({
        data: {
          _openid: openid,
          userInfo: {
            nickName: '',
            avatarUrl: ''
          },
          role: null,  // 待选择
          createdAt: new Date(),
          lastLoginAt: new Date()
        }
      });

      userData = {
        _id: createRes._id,
        _openid: openid,
        role: null
      };
    } else {
      // 老用户，更新最后登录时间
      userData = userRes.data[0];

      await db.collection('users').doc(userData._id).update({
        data: {
          lastLoginAt: new Date()
        }
      });
    }

    return {
      openid: openid,
      userInfo: userData.userInfo || {},
      role: userData.role
    };
  } catch (err) {
    console.error('login error:', err);
    throw err;
  }
};
```

### 2. updateUserRole 云函数

```javascript
// cloudfunctions/updateUserRole/index.js
const cloud = require('wx-server-sdk');
cloud.init({ env: cloud.DYNAMIC_CURRENT_ENV });

const db = cloud.database();

exports.main = async (event, context) => {
  const { role, roleInfo } = event;
  const wxContext = cloud.getWXContext();
  const openid = wxContext.OPENID;

  try {
    // 构造更新数据
    const updateData = {
      role: role,
      updatedAt: new Date()
    };

    if (role === 'student') {
      updateData.studentInfo = {
        grade: roleInfo.grade,
        school: roleInfo.school || '',
        realName: roleInfo.realName || '',
        studentId: roleInfo.studentId || ''
      };
    } else if (role === 'teacher') {
      updateData.teacherInfo = {
        school: roleInfo.school,
        subject: roleInfo.subject || '语文',
        realName: roleInfo.realName,
        phone: roleInfo.phone || '',
        certificationStatus: 'approved',  // 暂时自动通过
        certificationImages: roleInfo.certificationImages || [],
        certifiedAt: new Date()
      };
    }

    // 更新用户记录
    await db.collection('users').where({
      _openid: openid
    }).update({
      data: updateData
    });

    // 返回更新后的用户信息
    const userRes = await db.collection('users').where({
      _openid: openid
    }).get();

    return userRes.data[0];
  } catch (err) {
    console.error('updateUserRole error:', err);
    throw err;
  }
};
```

## 六、注意事项

### 1. 隐私合规
- 获取用户信息前需要用户授权
- 从2023年起，微信要求使用 `wx.getUserProfile()` 而非 `wx.getUserInfo()`
- 需要在小程序中添加《用户协议》和《隐私政策》

### 2. 用户体验
- 登录流程尽量简化
- 自动登录，避免重复登录
- 提供"退出登录"功能
- 登录失败时给出明确提示

### 3. 安全性
- openid 仅在云函数中获取，不暴露给前端
- 敏感信息加密存储
- 云函数做好权限验证

### 4. 错误处理
- 网络错误：提示用户检查网络
- 云函数错误：记录日志，给出友好提示
- 数据库错误：重试机制
